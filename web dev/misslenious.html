struct Student {
    char name[50];
    long long roll;
    char class[10];
    struct Student* next;
};

int main() {
    struct Student* first = (struct Student*) malloc(sizeof(struct Student));
    strcpy(first->name, "Vivek Yadav");
    first->roll = 51217702724;
    strcpy(first->class, "CSE A");

    struct Student* second = (struct Student*) malloc(sizeof(struct Student));
    strcpy(second->name, "Sakshar Mishra");
    second->roll = 36317702724;
    strcpy(second->class, "CSE B");

    first->next = second;
    second->next = NULL;

    struct Student* temp = first;
    while (temp != NULL) {
        printf("Name   : %s\n", temp->name);
        printf("Roll No: %lld\n", temp->roll);
        printf("Class  : %s\n\n", temp->class);
        temp = temp->next;
